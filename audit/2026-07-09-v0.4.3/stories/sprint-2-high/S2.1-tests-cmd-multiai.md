Créer `main_test.go` :

```go
func TestGetProfilesDir_ExplicitVar(t *testing.T) {
    t.Setenv("MULTIAI_PROFILES_DIR", "/custom/profiles")
    got := getProfilesDir()
    if got != "/custom/profiles" {
        t.Errorf("expected /custom/profiles, got %s", got)
    }
}

func TestGetProfilesDir_DevModeCWD(t *testing.T) {
    t.Setenv("MULTIAI_DEV", "1")
    // Ne pas set MULTIAI_PROFILES_DIR
    got := getProfilesDir()
    if !strings.Contains(got, "configs") || !strings.Contains(got, "profiles") {
        t.Errorf("expected CWD-relative configs/profiles, got %s", got)
    }
}

func TestGetProfilesDir_DevModeNotSet_SkipsCWD(t *testing.T) {
    // Sans MULTIAI_DEV, même si configs/profiles existe dans le CWD, on ne doit pas l'utiliser
    // Ce test vérifie le comportement de sécurité
    got := getProfilesDir()
    // Doit être le UserConfigDir, pas le CWD
    if strings.HasPrefix(got, ".") || filepath.IsAbs(got) && !strings.Contains(got, "multiai") {
        t.Logf("got: %s", got)
    }
}
```

### Test 4 : `ensureProfiles()`

```go
func TestEnsureProfiles_EmptyDir(t *testing.T) {
    dir := t.TempDir()
    t.Setenv("MULTIAI_PROFILES_DIR", dir)
    err := ensureProfiles(dir)
    if err != nil {
        t.Fatal(err)
    }
    files, _ := os.ReadDir(dir)
    if len(files) < 37 {
        t.Errorf("expected >=37 profiles, got %d", len(files))
    }
}

func TestEnsureProfiles_ExistingProfiles(t *testing.T) {
    dir := t.TempDir()
    // Créer un fichier pour simuler une installation existante
    os.WriteFile(filepath.Join(dir, "00-claude-official.env"), []byte("KEY=val"), 0644)
    t.Setenv("MULTIAI_PROFILES_DIR", dir)
    err := ensureProfiles(dir)
    if err != nil {
        t.Fatal(err)
    }
    // Le fichier existant ne doit pas être écrasé
    content, _ := os.ReadFile(filepath.Join(dir, "00-claude-official.env"))
    if string(content) != "KEY=val" {
        t.Error("existing profile was overwritten")
    }
}
```
