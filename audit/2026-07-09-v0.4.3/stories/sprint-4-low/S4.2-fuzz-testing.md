## Ce qu'il faut faire

### Fuzz 1 : Parseur .env

```go
// pkg/dotenv/fuzz_test.go
func FuzzParse(f *testing.F) {
    // Seeds
    f.Add("KEY=value")
    f.Add("KEY=\"quoted value\"")
    f.Add("KEY='single quoted'")
    f.Add("export KEY=value")
    f.Add("# comment\nKEY=value")
    f.Add("KEY=value with $VAR and %VAR%")
    f.Add("")
    f.Add("KEY=value\nKEY2=value2\nKEY3=value3")
    
    f.Fuzz(func(t *testing.T, input string) {
        // Ne doit jamais paniquer
        p, err := Parse(strings.NewReader(input))
        if err != nil {
            return // erreurs de parsing OK
        }
        // Vérifier que Get ne panique pas
        for _, key := range []string{"KEY", "KEY2", "", "NONEXISTENT"} {
            _ = p.Get(key)
        }
        // Vérifier que ListKeys ne panique pas
        _ = p.ListKeys()
    })
}
```

### Fuzz 2 : Parseur YAML

```go
// internal/profile/yaml_fuzz_test.go
func FuzzParseYAML(f *testing.F) {
    f.Add([]byte("id: test\ntool: claude\ncommand: claude\n"))
    f.Add([]byte("id: test\ntool: claude\nenv:\n  KEY: value\n"))
    f.Add([]byte(""))
    f.Add([]byte("invalid: [{"))
    
    f.Fuzz(func(t *testing.T, input []byte) {
        // Ne doit jamais paniquer
        prof, err := yamlToProfile(input)
        if err != nil {
            return
        }
        if prof != nil {
            // Vérifier que les champs obligatoires sont présents
            _ = prof.ID
            _ = prof.Tool
            _ = prof.Command
        }
    })
}
```

### Fuzz 3 : Expansion %VAR%

```go
// internal/env/env_fuzz_test.go
func FuzzExpandProfileEnv(f *testing.F) {
    f.Add("KEY=%HOME%", "HOME", "/home/user")
    f.Add("KEY=%NONEXISTENT%", "OTHER", "val")
    f.Add("%A%%B%%C%", "A", "1")
    
    f.Fuzz(func(t *testing.T, input string, varName string, varValue string) {
        // Éviter les noms de variables invalides
        if !regexp.MustCompile(`^[A-Z_][A-Z0-9_]*$`).MatchString(varName) {
            return
        }
        env := map[string]string{varName: varValue}
        // Ne doit jamais paniquer
        _ = ExpandProfileEnv(input, env, 10)
    })
}
```

### Étape 4 — Exécuter les fuzzers

```bash
cd multiai-go

go test -fuzz=FuzzParseYAML -fuzztime=30s ./internal/profile/
go test -fuzz=FuzzExpandProfileEnv -fuzztime=30s ./internal/env/
```
